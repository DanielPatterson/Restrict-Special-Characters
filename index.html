The HTML

The Javascript

function valid(f) {
	//requires: onkeyup="valid(this)" onblur="valid(this)"
	//the following is still able to be entered:
	// ` \u0060 | `
	// # \u0023 | #
	// ^ \u005E | ^
	// & \u0026 | &
	// ; \u003B | ; 

	!(/^[A-zÑñ0-9]*$/i).test(f.value)?f.value = f.value.replace(/[^A-zÑñ0-9]/ig,''):null;	
}
